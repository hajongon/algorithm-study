class Stack {
  constructor() {
    this.storage = {}
    this.last = -1
  }

  size() {
    return this.last + 1
  }

  push(num) {
    this.last++
    this.storage[this.last] = num
  }

  pop() {
    if (this.last === -1) return -1

    let pop = this.storage[this.last]
    delete this.storage[this.last]
    this.last--
    return pop
  }

  empty() {
    return this.last === -1 ? 1 : 0
  }

  top() {
    return this.last === -1 ? -1 : this.storage[this.last]
  }
}

    case 'size':
      answer.push(s.size())
      break
    case 'empty':
      answer.push(s.empty())
      break
    case 'pop':
      answer.push(s.pop())
      break
    default:
      break
  }
}
