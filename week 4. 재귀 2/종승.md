## 1780번 - 종이의 개수

```javascript
const fs = require("fs");
const input = fs.readFileSync("/dev/stdin").toString().trim().split("\n");

const length = +input[0];
const d = Math.floor(length / 3);

const arr = input
  .filter((el, idx) => idx !== 0)
  .map((el) => el.split(" ").map((el) => +el));

const numberCount = [0, 0, 0];

const count = (array, r, c, d) => {
  // d가 1이면 모든 요소 카운트
  if (d === 1) {
    numberCount[array[r][c] + 1]++;
    return;
  }
  // 기준점의 요소를 first에 할당
  const first = array[r][c];
  let isSame = true;
  // 행 기준점으로 부터 +d 칸
  for (let i = r; i < r + d; i++) {
    // 열 기준점으로 부터 +d 칸
    for (let j = c; j < c + d; j++) {
      // 다른 걸 발견하면
      if (array[i][j] !== first) {
        isSame = false;
        // 열 반복문 중단
        break;
      }
    }
    // 다른 걸 발견하면
    if (!isSame) {
      // 행 반복문 중단
      break;
    }
  }

  // 모두 같으면
  if (isSame) {
    // 해당 인덱스 ++
    numberCount[first + 1]++;
    // 하나라도 다른 것이 있으면
  } else {
    // d / 3
    const newD = Math.floor(d / 3);
    // 기준점을 9개 만들어서 재귀 9번
    for (let i = 0; i < 3; i++) {
      for (let j = 0; j < 3; j++) {
        // 9등분할 필요없이 반복분 돌리는 기준점을 바꿔가면서 재귀호출!!
        count(array, r + i * newD, c + j * newD, newD);
      }
    }
  }
};

count(arr, 0, 0, length);

for (let el of numberCount) {
  console.log(el);
}
```
