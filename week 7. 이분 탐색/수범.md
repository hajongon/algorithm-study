# 1822. 차집합

```javascript
function solution() {
  /*
    문제 : A집합에서 B에 속하지않는(차집합) 원소를 구하는 프로그램을 작성해라.
    의사코드
    1.set 이라는 테이블에 B집합을 할당한다.
    2.set에 A집합의 요소를 순차적으로 추가한다.
    3.set은 같은 원소를 허용하지 않기때문에, 같은원소가 추가된다면 삭제된다.
    4.set에 A집합의 요소가 추가된다면 B집합에 속하지 않는 요소이기에 result배열에 푸쉬해준다.
    5.result 배열에는 B집합에 속하지않는 요소만 모여있을테니, 출력해준다.
    6.result 배열이 비어있다면 차집합이 없는거니까 0을 출력한다.
     */
  const fs = require("fs");
  const filePath = process.platform === "linux" ? "/dev/stdin" : "./input.txt";
  const [A, B] = fs
    .readFileSync(filePath)
    .toString()
    .trim()
    .split("\n")
    .slice(1);
  let set = new Set(B.split(" "));
  let result = [];
  for (let el of A.trim().split(" ")) {
    let size = set.size;
    set.add(el);
    if (set.size > size) {
      result.push(Number(el));
    }
  }
  result.sort((a, b) => a - b);
  console.log(
    result.length === 0 ? 0 : `${result.length}\n` + result.join(" ")
  );
}
solution();
```

# 10815. 숫자카드

```javascript
function solution() {
  /*

       */
  const fs = require("fs");
  const filePath = process.platform === "linux" ? "/dev/stdin" : "./input.txt";
  const input = fs.readFileSync(filePath).toString().trim().split("\n");
  let get = input[1]
    .split(" ")
    .map(Number)
    .sort((a, b) => a - b);
  let find = input[3].split(" ").map(Number);
  let set = new Set(get);
  let result = [];
  let size = get.length;
  for (let el of find) {
    set.add(el);
    if (set.size === size) {
      // 가지고있는 카드인경우
      result.push(1);
    } else if (set.size > size) {
      //가지고있지 않은 경우
      size++;
      result.push(0);
    }
  }
  console.log(...result);
}
solution();
```
